# Setup the project
cmake_minimum_required(VERSION 2.6 FATAL_ERROR)
set(CMAKE_BUILD_TYPE Debug)
project(CRTest)

  set(ROOT_DIR ${ROOTSYS}/cmake)
endif()
find_package(ROOT REQUIRED)

include(${Geant4_USE_FILE}
     ${ROOT_USE_FILE})
# Locate source and headers for project
